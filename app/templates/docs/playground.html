import requests

url = "http://localhost:5000/api/v1/create-payment"
headers = {
    "Authorization": "Bearer {{ active_key.public_key if active_key else 'YOUR_KEY' }}",
    "Content-Type": "application/json"
}
data = {
    "amount": 100.00,
    "upi_address": "merchant@upi",
    "target_callback_url": "https://webhook.site/test",
    "return_url": "http://localhost:5000/dashboard"
}

response = requests.post(url, json=data, headers=headers)
print(response.json())
<?php
$ch = curl_init('http://localhost:5000/api/v1/create-payment');
$payload = json_encode([
    'amount' => 100.00,
    'upi_address' => 'merchant@upi',
    'target_callback_url' => 'https://webhook.site/test',
    'return_url' => 'http://localhost:5000/dashboard'
]);
curl_setopt($ch, CURLOPT_POSTFIELDS, $payload);
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    'Authorization: Bearer {{ active_key.public_key if active_key else "YOUR_KEY" }}',
    'Content-Type: application/json'
]);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
$response = curl_exec($ch);
curl_close($ch);
echo $response;
?>
